#!/bin/bash
# YesQuid Task Worker - STRICT, NO-ILLUSION MODE
# Processes ONE task file and only marks it done if a real artifact exists.
set -euo pipefail
PLAN_ID="${1:-52f7b763}"
TASKS_ROOT="$HOME/tasks"
ACTIVE_DIR="$TASKS_ROOT/active/plan_${PLAN_ID}"
DONE_DIR="$TASKS_ROOT/done/plan_${PLAN_ID}"
OUTPUT_ROOT="$HOME/sovereign-architect/storage/outputs"
LOG_DIR="$OUTPUT_ROOT/logs"
ART_ROOT="$HOME/sovereign-architect/storage/artifacts/plan_${PLAN_ID}"
mkdir -p "$DONE_DIR" "$LOG_DIR" "$ART_ROOT"
if [ ! -d "$ACTIVE_DIR" ]; then
  echo "No active task directory at $ACTIVE_DIR"
  exit 0
fi
TASK_FILE="$(ls "$ACTIVE_DIR"/*.json 2>/dev/null | head -n 1 || true)"
if [ -z "${TASK_FILE}" ]; then
  echo "No tasks left in $ACTIVE_DIR"
  exit 0
